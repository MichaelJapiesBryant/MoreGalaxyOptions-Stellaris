fallen_machine_empire_1 = {
	graphical_culture = "fallen_machine_empire_01"
	initializer = "fallen_machine"

	weight_modifier = {
		base = 100
	}
	
	possible = {
		host_has_dlc = "Synthetic Dawn Story Pack"
	}
	
	create_country_effect = {
		create_species = {
			name = random
			class = MACHINE
			namelist = MACHINE3
			portrait = random
			traits = {
				trait = trait_machine_unit
				trait = random_traits
			}
			extra_trait_points = 3
			allow_negative_traits = no
		}
		create_country = {
			name = random
			type = fallen_empire
			ignore_initial_colony_error = yes
			authority = auth_machine_intelligence
			civics = {
				civic = civic_machine_remnants
				civic = civic_custodian_directives
			}
			species = last_created
			ethos = {
				ethic = ethic_gestalt_consciousness
			}			
			flag = random
			origin = "origin_fallen_empire"
			effect = {
				set_country_flag = fallen_empire_machine
				add_resource = {
					minerals = 10000
					energy = 10000	
					food = 1000
					influence = 500
				}			
				if = {
					limit = { host_has_dlc = "Apocalypse" }
					give_technology = {
						tech = tech_colossus
						message = no
					}
				}
				# must initialize global designs here
				add_global_ship_design = "NAME_Omega"
				add_global_ship_design = "NAME_Alpha"
				add_global_ship_design = "NAME_Beta"
				add_global_ship_design = "NAME_Gamma"
				add_global_ship_design = "NAME_Theta"
				add_global_ship_design = "NAME_Tau"
				add_global_ship_design = "NAME_Sigma"
				add_global_ship_design = "NAME_FE_Starbase"
			}
		}
	}

	graphical_culture = "fallen_machine_empire_01"
	initializer = "fallen_machine"

	weight_modifier = {
		base = 100
	}
	
	possible = {
		host_has_dlc = "Synthetic Dawn Story Pack"
	}
	
	create_country_effect = {
		create_species = {
			name = random
			class = MACHINE
			namelist = MACHINE3
			portrait = random
			traits = {
				trait = trait_machine_unit
				trait = random_traits
			}
			extra_trait_points = 3
			allow_negative_traits = no
		}
		create_country = {
			name = random
			type = fallen_empire
			ignore_initial_colony_error = yes
			authority = auth_machine_intelligence
			civics = {
				civic = civic_machine_remnants
				civic = civic_custodian_directives
			}
			species = last_created
			ethos = {
				ethic = ethic_gestalt_consciousness
			}			
			flag = random
			origin = "origin_fallen_empire"
			effect = {
				set_country_flag = fallen_empire_machine
				add_resource = {
					minerals = 10000
					energy = 10000	
					food = 1000
					influence = 500
				}			
				if = {
					limit = { host_has_dlc = "Apocalypse" }
					give_technology = {
						tech = tech_colossus
						message = no
					}
				}
				# must initialize global designs here
				add_global_ship_design = "NAME_Omega"
				add_global_ship_design = "NAME_Alpha"
				add_global_ship_design = "NAME_Beta"
				add_global_ship_design = "NAME_Gamma"
				add_global_ship_design = "NAME_Theta"
				add_global_ship_design = "NAME_Tau"
				add_global_ship_design = "NAME_Sigma"
				add_global_ship_design = "NAME_FE_Starbase"
			}
		}
	}
}
